Define Outcome Measure Functions

stand_bias = function(theta_hat=NULL, theta=NULL){
  if (is.data.frame(theta_hat)){
    if (ncol(theta_hat) != length(theta)){ stop('Mismatch between the number of theta_hat columns and length theta.') }
    SB = matrix(sapply(1:ncol(theta_hat), function(x) (mean(theta_hat[,x]) - theta[x]) / sd(theta_hat[,x])), ncol=length(theta_hat))
    colnames(SB) = colnames(theta_hat); rownames(SB) = 'SB'
  } else {
    SB = (mean(theta_hat) - theta) / sd(theta_hat)
  }
  return(SB)
}
rmse = function(theta_hat=NULL, theta=NULL){
  if (is.data.frame(theta_hat)){
    if (ncol(theta_hat) != length(theta)){ stop('Mismatch between the number of theta_hat columns and length theta.') }
    rmse = matrix(sapply(1:ncol(theta_hat), function(x) sqrt(sum((theta_hat[,x] - theta[x])^2) / length(theta_hat[,x]))), ncol=length(theta_hat))
    colnames(rmse) = colnames(theta_hat); rownames(rmse) = 'rmse'
  } else {
    rmse = sqrt(sum((theta_hat - theta)^2) / length(theta_hat))
  }
  return(rmse)
}

generate_multigrowth = function(Nscans = 0, design = NULL, missing = 0,
                                agebeta = 0, expbeta = 0, intbeta = 0,
                                age2beta = 0, indsd = .5, perfsd = 1){
  
  fit = misfit = NA
  while ((!isS4(misfit) & !is.list(misfit)) | (!isS4(fit) & !is.list(fit))){
    centerObs = 1
    wavespace = 1
    if (design == 'accelerated'){
      Obs = 3
      N = round(Nscans / Obs)
      meanage = 14
      ageT1 = rnorm(N, mean=meanage, sd=2)
    } else if (design == 'cohort'){
      Obs = 5
      N = round(Nscans / Obs)
      firstage = 12
      meanage = 14
      ageT1 = rnorm(N, mean=firstage, sd = .00)
    } else if (design == 'cross.sectional'){
      Obs = 1
      N = round(Nscans / Obs)
      meanage = 14
      ageT1 = rnorm(N, mean=meanage, sd=2)
    }
    dat = data.frame(id = factor(), age = double(), exposure = double(), performance = double())
    # Generate Repeated Measures
    for (sub in 1:N){
      idvar = rnorm(1, mean=0, sd=indsd)
      for (t in 1:Obs){
        if (sample(1:100,1) <= missing*100){
          temp = data.frame(id = sub, age = NA, exposure = NA, performance = NA)
        } else {
          age = (ageT1[sub] - meanage) + (t - 1)*wavespace + rnorm(1, mean=0, sd=.1)
          exp = (t - centerObs)
          perform = 2 + idvar + agebeta*age + expbeta*exp + intbeta*(age*exp) + age2beta*(age^2) + rnorm(1, mean=0, sd=perfsd)
          temp = data.frame(id = sub, age = age, exposure = exp, performance = perform)
        }
        dat = rbind(dat,temp)
      }
    }
    # Deal with Scaling Issues
    dat[,'performance'] = scale(dat[,'performance'], center=TRUE, scale=FALSE)
    dat[,'performance'] = dat[,'performance'] - min(dat[,'performance'], na.rm=TRUE) + 1
  
    if (design == 'cross.sectional'){
      fit = misfit = tryCatch(lm(performance ~ age, data=dat),
                              warning=function(w){NA},
                              finally=NA)
    } else {
      misfit = tryCatch(lmer(performance ~ age + (1 | id), data=dat),
                        warning=function(w){NA},
                        finally=NA)
      fit = tryCatch(lmer(performance ~ age + exposure + (1 | id), data=dat),
                     warning=function(w){NA},
                     finally=NA)
    }
  }
  # Fit Models
  if (design == 'cross.sectional'){
    intergrowth.cor = NA
    varinflation = NA
    misfit = NA
    fixed.coeffs.misfit = NA
    var.est.misfit = NA
    fixed.coeffs.fit = summary(fit)$coefficients
    var.est.fit = sigma(fit)
  } else {
    intergrowth.cor = cor(dat$age, dat$exposure, use='complete.obs')
    varinflation = car::vif(lm(performance ~ age + exposure, data=dat))
    fixed.coeffs.misfit = summary(misfit)$coefficients
    var.est.misfit = summary(misfit)$varcor
    fixed.coeffs.fit = summary(fit)$coefficients
    var.est.fit = summary(fit)$varcor
  }
  output = list(
    intergrowth.cor = intergrowth.cor,
    varinflation = varinflation,
    dat = dat,
    misfit = misfit,
    fixed.coeffs.misfit = fixed.coeffs.misfit,
    var.est.misfit = var.est.misfit,
    fit = fit,
    fixed.coeffs.fit = fixed.coeffs.fit,
    var.est.fit = var.est.fit
  )
  return(output)
}

for (i in 1:nrow(des.matrix)){
  # Data Generation
  templist = list()
  for (j in 1:iter){
    templist[[j]] = list()
    names(templist)[j] = paste0('iter',j)

                                         missing = des.matrix[i,'missing'], agebeta = des.matrix[i,'agebeta'],
                                         expbeta = des.matrix[i,'expbeta'], intbeta = des.matrix[i,'intbeta'],

  # Extract Model Information
  if (des.matrix[i,'design'] != 'cross.sectional'){
    fix.misfit = sapply(templist, '[[', 'fixed.coeffs.misfit', simplify = FALSE)
    var.misfit = sapply(templist, '[[', 'var.est.misfit', simplify = FALSE)
    fix.fit = sapply(templist, '[[', 'fixed.coeffs.fit', simplify = FALSE)
    var.fit = sapply(templist, '[[', 'var.est.fit', simplify = FALSE)

                         var.inflate = sapply(templist, '[[', 'varinflation')[1,],
                         misfit.agebeta = sapply(fix.misfit, function(x) x['age','Estimate']),                               
                         misfit.agesig = sapply(fix.misfit, function(x) x['age','Pr(>|t|)']),
                         misfit.idvar = sapply(var.misfit, function(x) sqrt(x$id)),
                         misfit.residvar = sapply(var.misfit, function(x) attr(x, 's')),
                         
                         fit.agebeta = sapply(fix.fit, function(x) x['age','Estimate']),
                         fit.agesig = sapply(fix.fit, function(x) x['age','Pr(>|t|)']),
                         fit.expbeta = sapply(fix.fit, function(x) x['exposure','Estimate']),
                         fit.expsig = sapply(fix.fit, function(x) x['exposure','Pr(>|t|)']),
                         fit.idvar = sapply(var.fit, function(x) sqrt(x$id)),
                         fit.residvar = sapply(var.fit, function(x) attr(x, 's'))
    )
  } else {
    fix.fit = sapply(templist, '[[', 'fixed.coeffs.fit', simplify = FALSE)
    var.fit = sapply(templist, '[[', 'var.est.fit', simplify = FALSE)
    outlist = data.frame(fit.agebeta = sapply(fix.fit, function(x) x['age','Estimate']),
                         fit.agesig = sapply(fix.fit, function(x) x['age','Pr(>|t|)']),
                         fit.residvar = sapply(templist, '[[', 'var.est.fit', simplify = TRUE)
    )
  }
  
  # Save Data
  design = des.matrix[i,]
  numgendat = sum(str_count(string = des.matrix[1:i,'design'], pattern = paste0(des.matrix[i,'design'])))
  save(design, templist, outlist,
       file = paste0(outputdir,des.matrix[i,'design'],'.gendata.', numgendat,'.Rdata'),
       compress = 'xz')
  
  rm(design, templist, outlist)
}

## Data Generation Function 2
generate_multigrowth2 = function(Nscans = 0, design = NULL, missing = 0,
                                 agebeta = 0, expbeta = 0, intbeta = 0,
                                 age2beta = 0, indsd = .5, perfsd = 1){
  
  fit = misfit = NA
  while ((!isS4(misfit) & !is.list(misfit)) | (!isS4(fit) & !is.list(fit))){
    centerObs = 1
    wavespace = 1
    if (design == 'accelerated'){
      Obs = 3
      N = round(Nscans / Obs)
      meanage = 14
      ageT1 = rnorm(N, mean=meanage, sd=2)
    } else if (design == 'cohort'){
      Obs = 5
      N = round(Nscans / Obs)
      firstage = 12
      meanage = 14
      ageT1 = rnorm(N, mean=firstage, sd = .00)
    } else if (design == 'cross.sectional'){
      Obs = 1
      N = round(Nscans / Obs)
      meanage = 14
      ageT1 = rnorm(N, mean=meanage, sd=2)
    }
    dat = data.frame(id = factor(), age = double(), exposure = double(), performance = double())
    # Generate Repeated Measures
    for (sub in 1:N){
      idvar = rnorm(1, mean=0, sd=indsd)
      for (t in 1:Obs){
        if (sample(1:100,1) <= missing*100){
          temp = data.frame(id = sub, age = NA, exposure = NA, performance = NA)
        } else {
          age = (ageT1[sub] - meanage) + (t - 1)*wavespace + rnorm(1, mean=0, sd=.1)
          exp = (t - centerObs)
          perform = 2 + idvar + agebeta*age + expbeta*exp + intbeta*(age*exp) + age2beta*(age^2) + rnorm(1, mean=0, sd=perfsd)
          temp = data.frame(id = sub, age = age, exposure = exp, performance = perform)
        }
        dat = rbind(dat,temp)
      }
    }
    # Deal with Scaling Issues
    dat[,'performance'] = scale(dat[,'performance'], center=TRUE, scale=FALSE)
    dat[,'performance'] = dat[,'performance'] - min(dat[,'performance'], na.rm=TRUE) + 1
    
    if (design == 'cross.sectional'){
      fit = tryCatch(lm(performance ~ age + I(age^2), data=dat),
                     warning=function(w){NA},
                     finally=NA)
      misfit = fit
    } else {
      misfit = tryCatch(lmer(performance ~ age + I(age^2) + (1 | id), data=dat),
                        warning=function(w){NA},
                        finally=NA)
      fit = tryCatch(lmer(performance ~ age + I(age^2) + exposure + (1 | id), data=dat),
                     warning=function(w){NA},
                     finally=NA)
    }
  }
  # Fit Models
  if (design == 'cross.sectional'){
    intergrowth.cor = NA
    varinflation = NA
    misfit = NA
    fixed.coeffs.misfit = NA
    var.est.misfit = NA
    fixed.coeffs.fit = summary(fit)$coefficients
    var.est.fit = sigma(fit)
  } else {
    intergrowth.cor = cor(dat$age, dat$exposure)
    varinflation = car::vif(lm(performance ~ age + exposure, data=dat))
    fixed.coeffs.misfit = summary(misfit)$coefficients
    var.est.misfit = summary(misfit)$varcor
    fixed.coeffs.fit = summary(fit)$coefficients
    var.est.fit = summary(fit)$varcor
  }
  output = list(
    intergrowth.cor = intergrowth.cor,
    varinflation = varinflation,
    dat = dat,
    misfit = misfit,
    fixed.coeffs.misfit = fixed.coeffs.misfit,
    var.est.misfit = var.est.misfit,
    fit = fit,
    fixed.coeffs.fit = fixed.coeffs.fit,
    var.est.fit = var.est.fit
  )
  return(output)
}

set.seed(19728)
Nscans = 750
iter = 1000

generate_puberty = function(Nscans = 0, design = NULL, missing = 0,
                            agebeta = 0, pubbeta = 0, intbeta = 0,
                            pubvar = NULL, samplerate = NULL,
                            indsd = .5, rewardsd = .5){
  
  fit = misfit = tanfit = NA
  while ((!isS4(misfit) & !is.list(misfit)) | (!isS4(fit) & !is.list(fit)) | (!isS4(tanfit) & !is.list(tanfit))){
    
    meanage = 12
    wavespace = ifelse(samplerate == 'fast', .5, 1)
    if (design == 'accelerated'){
      Obs = 3
      N = round(Nscans / Obs)
      firstage = ifelse(samplerate == 'fast', 11.5, 11)
      ageT1 = rnorm(N, mean=firstage, sd=1)
      centerObs = median(1:Obs)
    } else if (design == 'cohort'){
      Obs = 4
      N = round(Nscans / Obs)
      firstage = ifelse(samplerate == 'fast', 11.25, 10.5)
      ageT1 = rnorm(N, mean=firstage, sd=0)
      centerObs = median(1:Obs)
    } else if (design == 'cross.sectional'){
      Obs = 1
      N = round(Nscans / Obs)
      ageT1 = rnorm(N, mean=meanage, sd=2)
    }
    
    dat = data.frame(id = factor(), wave = double(), age = double(), puberty = double(), tanner = double(), risk = double())
    
    # Set Up Puberty Parameters
    x = seq(-4, 4, .1)
    q = matrix(nrow=length(x), ncol=N)
    colnames(q) = 1:ncol(q)
    
    z = mvrnorm(N, mu=c(0,0), Sigma=matrix(c(1,.3,.3, 1), nrow=2), empirical=FALSE)
    u = pnorm(z)
    if (pubvar == 'high'){
      location = qtruncnorm(u[,1], a=-3, b=3, mean=0, sd=1.5)
    } else if (pubvar == 'low'){
      location = qtruncnorm(u[,1], a=-3, b=3, mean=0, sd=.5)
    }
    
    shape = qgamma(u[,2], 10, 3)
    
    # Generate Repeated Measures
    for (sub in 1:N){
      
      # Generate Individual Puberty Curve
      q[,sub] = 1/(1+exp(-1*shape[sub]*(x-location[sub])))
      
      idvar = rnorm(1, mean=0, sd=indsd)
      for (t in 1:Obs){
        if (sample(1:100,1) <= missing*100){
          temp = data.frame(id=sub, wave=NA, age=NA, puberty=NA, tanner = NA, reward = NA)
        } else {
          age = (ageT1[sub] - meanage) + (t - 1)*wavespace + rnorm(1, mean=0, sd=.1)
          pubvalue = q[which(abs(x-age)==min(abs(x-age))), sub]
          if (pubvalue < .05){tan = 1
          } else if (pubvalue < .35){tan = 2
          } else if (pubvalue < .65){tan = 3
          } else if (pubvalue < .95){tan = 4
          } else {tan = 5}
          pubvalue = pubvalue - .5
          rw = 2 + idvar + agebeta*age + pubbeta*pubvalue + intbeta*(age*pubvalue) + rnorm(1, mean=0, sd=rewardsd)
          temp = data.frame(id=sub, wave=t, age=age, puberty=pubvalue, tanner = tan, reward = rw)
        }
        dat = rbind(dat,temp)
      }
    }
    # Deal with Scaling Issues
    dat[,'reward'] = scale(dat[,'reward'], center=TRUE, scale=FALSE)
    dat[,'reward'] = dat[,'reward'] - min(dat[,'reward'], na.rm=TRUE)
    
    if (design == 'cross.sectional'){
      fit = tryCatch(lm(reward ~ age + I(age^2), data=dat),
                     warning=function(w){NA},
                     finally=NA)
      misfit = tanfit = fit
    } else {
      misfit = tryCatch(lmer(reward ~ age + I(age^2) + (1 | id), data=dat),
                        warning=function(w){NA},
                        finally=NA)
      fit = tryCatch(lmer(reward ~ age + puberty + age:puberty + (1 | id), data=dat),
                     warning=function(w){NA},
                     finally=NA)
      tanfit = tryCatch(lmer(reward ~ age + tanner + age:tanner + (1 | id), data=dat),
                        warning=function(w){NA},
                        finally=NA)
    }
  }
  # Fit Models
  if (design == 'cross.sectional'){
    intergrowth.cor = NA
    varinflation = NA
    misfit = NA
    fixed.coeffs.misfit = NA
    var.est.misfit = NA
    fixed.coeffs.fit = summary(fit)$coefficients
    var.est.fit = sigma(fit)
    fixed.coeffs.tanfit = summary(tanfit)$coefficients
    var.est.tanfit = sigma(tanfit)
  } else {
    intergrowth.cor = cor(dat$age, dat$puberty)
    intergrowth.cor.tanner = cor(dat$age, dat$tanner)
    pub.tanner.cor = cor(dat$puberty, dat$tanner)
    varinflation = car::vif(lm(reward ~ age + puberty, data=dat))
    fixed.coeffs.misfit = summary(misfit)$coefficients
    var.est.misfit = summary(misfit)$varcor
    fixed.coeffs.fit = summary(fit)$coefficients
    var.est.fit = summary(fit)$varcor
    fixed.coeffs.tanfit = summary(tanfit)$coefficients
    var.est.tanfit = summary(fit)$varcor
  }
  output = list(
    intergrowth.cor = intergrowth.cor,
    intergrowth.cor.tanner = intergrowth.cor.tanner,
    pub.tanner.cor = pub.tanner.cor,
    varinflation = varinflation,
    dat = dat,
    misfit = misfit,
    fixed.coeffs.misfit = fixed.coeffs.misfit,
    var.est.misfit = var.est.misfit,
    fit = fit,
    fixed.coeffs.fit = fixed.coeffs.fit,
    var.est.fit = var.est.fit,
    tanfit = tanfit,
    fixed.coeffs.tanfit = fixed.coeffs.tanfit,
    var.est.tanfit = var.est.tanfit,
    q = q,
    x = x
  )
  return(output)
}
